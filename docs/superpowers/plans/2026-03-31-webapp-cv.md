# Webapp CV — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single-page scrollytelling CV webapp with a Three.js 3D morphing animation, sci-fi RPG aesthetic (HUD overlay, glitch effects, stat bars), and four narrative chapters, deployed to GitHub Pages via a `gh-pages` branch.

**Architecture:** A fixed `WebGLRenderer` canvas sits behind transparent HTML sections. Shared `state.js` holds `chapterIndex`, `chapterProgress`, and `totalProgress`; `scroll.js` computes these on each animation frame; `animation.js` lerps 120-vertex `BufferGeometry` between four target shapes (seed → lattice → helix → dispersed); `scene.js` ties everything together in a single render loop. No build step — Three.js via import map CDN, Tailwind via play CDN.

**Tech Stack:** HTML5, CSS3, Tailwind CSS (CDN), Three.js r170 (import map CDN), ES Modules, GitHub Pages

```javascript
import { state } from './state.js';

export function updateScroll() {
  const scrollY = window.scrollY;
  const vh      = window.innerHeight;
  // 4 chapters × 100vh; last chapter goes 0→1 as we scroll through it
  const scrollable = vh * 3;

  state.totalProgress  = Math.min(scrollY / scrollable, 1);
  const rawChapter     = scrollY / vh;
  state.chapterIndex   = Math.min(Math.floor(rawChapter), 3);
  state.chapterProgress = rawChapter - state.chapterIndex;
}
```

```javascript
import * as THREE from 'three';
import { EffectComposer }  from 'three/addons/postprocessing/EffectComposer.js';
import { RenderPass }      from 'three/addons/postprocessing/RenderPass.js';
import { UnrealBloomPass } from 'three/addons/postprocessing/UnrealBloomPass.js';
import { updateScroll } from './scroll.js';
import { initGeometry, updateMorph } from './animation.js';
import { updateHUD }    from './hud.js';
import './cursor.js';

// --- Scene ---
const scene  = new THREE.Scene();
const camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 100);
camera.position.set(0, 0, 5);

// --- Renderer ---
const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.setClearColor(0x000000, 0);
renderer.domElement.id = 'three-canvas';
document.body.prepend(renderer.domElement);

// --- Bloom composer ---
const composer = new EffectComposer(renderer);
composer.addPass(new RenderPass(scene, camera));
const bloomPass = new UnrealBloomPass(
  new THREE.Vector2(window.innerWidth, window.innerHeight),
  1.5,  // strength
  0.4,  // radius
  0.1   // threshold
);
composer.addPass(bloomPass);

// --- Resize ---
window.addEventListener('resize', () => {
  camera.aspect = window.innerWidth / window.innerHeight;
  camera.updateProjectionMatrix();
  renderer.setSize(window.innerWidth, window.innerHeight);
  composer.setSize(window.innerWidth, window.innerHeight);
});

// --- Lights (three orbiting point lights) ---
const LIGHT_DEFS = [
  { color: 0x00f5ff, distance: 4, phase: 0,                    speed:  0.4  }, // cyan
  { color: 0x8b5cf6, distance: 4, phase: (2 * Math.PI) / 3,   speed: -0.3  }, // violet, opposite direction
  { color: 0xff0080, distance: 4, phase: (4 * Math.PI) / 3,   speed:  0.5  }, // magenta
];

const pointLights = LIGHT_DEFS.map(({ color, distance, phase }) => {
  const light = new THREE.PointLight(color, 3, 10);
  light.position.set(distance * Math.cos(phase), Math.sin(phase) * 1.5, distance * Math.sin(phase));
  scene.add(light);
  return light;
});

// --- Geometry init ---
initGeometry(scene);

// --- IntersectionObserver for chapter animations ---
const chapterSections = document.querySelectorAll('.chapter');
const observer = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      entry.target.classList.add('active');
      const glitch = entry.target.querySelector('.glitch-target');
      if (glitch) {
        glitch.classList.remove('glitch-in');
        void glitch.offsetWidth; // force reflow to restart animation
        glitch.classList.add('glitch-in');
      }
    }
  });
}, { threshold: 0.35 });
chapterSections.forEach(s => observer.observe(s));

// --- Render loop ---
let active = true;
document.addEventListener('visibilitychange', () => { active = !document.hidden; });

function animate() {
  requestAnimationFrame(animate);
  if (!active) return;

  updateScroll();
  updateMorph();
  updateHUD();

  // Orbit lights
  const t = performance.now() * 0.001;
  LIGHT_DEFS.forEach(({ distance, phase, speed }, i) => {
    const a = phase + t * speed;
    pointLights[i].position.set(
      distance * Math.cos(a),
      Math.sin(a * 0.7) * 2,
      distance * Math.sin(a)
    );
  });

  composer.render();
}
animate();
```

### Task 4: 3D geometry + vertex morphing

```javascript
import * as THREE from 'three';
import { state }  from './state.js';

const N = 120; // vertex count

// --- Target shape generators ---

function makeSeedTargets() {
  const p = new Float32Array(N * 3);
  for (let i = 0; i < N; i++) {
    const r     = Math.random() * 0.4;
    const theta = Math.random() * Math.PI * 2;
    const phi   = Math.acos(2 * Math.random() - 1);
    p[i*3]   = r * Math.sin(phi) * Math.cos(theta);
    p[i*3+1] = r * Math.sin(phi) * Math.sin(theta);
    p[i*3+2] = r * Math.cos(phi);
  }
  return p;
}

function makeLatticeTargets() {
  // 5 × 4 × 6 = 120
  const p = new Float32Array(N * 3);
  let i = 0;
  for (let x = 0; x < 5; x++)
    for (let y = 0; y < 4; y++)
      for (let z = 0; z < 6; z++) {
        p[i*3]   = (x - 2)   * 0.9;
        p[i*3+1] = (y - 1.5) * 0.9;
        p[i*3+2] = (z - 2.5) * 0.9;
        i++;
      }
  return p;
}

function makeHelixTargets() {
  // 60 pts per strand × 2 strands
  const p       = new Float32Array(N * 3);
  const half    = N / 2;
  const radius  = 0.9;
  const pitch   = 3.0 / half;
  for (let s = 0; s < 2; s++) {
    const phase = s * Math.PI;
    for (let i = 0; i < half; i++) {
      const t   = (i / half) * Math.PI * 4;
      const idx = (s * half + i) * 3;
      p[idx]   = radius * Math.cos(t + phase);
      p[idx+1] = i * pitch - 1.5;
      p[idx+2] = radius * Math.sin(t + phase);
    }
  }
  return p;
}

function makeDispersedTargets() {
  const p = new Float32Array(N * 3);
  for (let i = 0; i < N; i++) {
    const r     = 3.5 + Math.random() * 2;
    const theta = Math.random() * Math.PI * 2;
    const phi   = Math.acos(2 * Math.random() - 1);
    p[i*3]   = r * Math.sin(phi) * Math.cos(theta);
    p[i*3+1] = r * Math.sin(phi) * Math.sin(theta);
    p[i*3+2] = r * Math.cos(phi);
  }
  return p;
}

// Pre-compute all targets once
const TARGETS = [
  makeSeedTargets(),
  makeLatticeTargets(),
  makeHelixTargets(),
  makeDispersedTargets(),
];

// Working buffer — starts at seed positions
const current = new Float32Array(TARGETS[0]);

// Vertex colours: cycle cyan / violet / magenta
function makeColors() {
  const c  = new Float32Array(N * 3);
  const palette = [[0, 0.96, 1], [0.54, 0.36, 0.96], [1, 0, 0.5]];
  for (let i = 0; i < N; i++) {
    const [r, g, b] = palette[i % 3];
    c[i*3] = r; c[i*3+1] = g; c[i*3+2] = b;
  }
  return c;
}

let geometry;
let points;

export function initGeometry(scene) {
  geometry = new THREE.BufferGeometry();
  geometry.setAttribute('position', new THREE.BufferAttribute(current, 3));
  geometry.setAttribute('color',    new THREE.BufferAttribute(makeColors(), 3));

  const material = new THREE.PointsMaterial({
    size: 0.07,
    vertexColors: true,
    blending: THREE.AdditiveBlending,
    transparent: true,
    depthWrite: false,
    sizeAttenuation: true,
  });

  points = new THREE.Points(geometry, material);
  scene.add(points);
}

export function updateMorph() {
  if (!geometry) return;

  const { chapterIndex, chapterProgress } = state;
  const from = TARGETS[chapterIndex];
  const to   = TARGETS[Math.min(chapterIndex + 1, 3)];

  for (let i = 0; i < current.length; i++) {
    current[i] = from[i] + (to[i] - from[i]) * chapterProgress;
  }

  geometry.attributes.position.needsUpdate = true;

  // Slow ambient rotation
  if (points) points.rotation.y += 0.002;
}
```

```javascript
const CHAPTER_NAMES = ['ORIGIN', 'CAPABILITIES', 'MISSION LOG', 'NEXT OBJECTIVE'];

const chapterEl  = document.getElementById('hud-chapter');

  if (chapterIndex !== lastChapter) {
    const num = String(chapterIndex + 1).padStart(2, '0');
    chapterEl.textContent = `[ CHAPTER ${num} / 04 — ${CHAPTER_NAMES[chapterIndex]} ]`;
    lastChapter = chapterIndex;

}
```
